Algoritmo DescobrirMinMax

MIN <- 0
MAX <- 0
FLAG <- 1
COUNT <- 0
Repeat until COUNT = 100
    Read(N)
    COUNT <- COUNT +1
    If FLAG = 1 Then 
        MIN <- N
        MAX <- N
        FLAG <- 0
    Else
      If N >= MAX Then 
        MAX <- N
      If N <= MIN Then
        MIN <- N
Write ("O minimo é " MIN, " e o máximo é", MAX)
Exit()
